------------------------------  

```
CREATE DATABASE test_db;
CREATE USER 'db-test_db'@'localhost' IDENTIFIED BY 'netlab!db!test_db';
GRANT ALL PRIVILEGES ON test_db.* TO 'db-test_db'@'localhost';
FLUSH PRIVILEGES;
QUIT;
```
